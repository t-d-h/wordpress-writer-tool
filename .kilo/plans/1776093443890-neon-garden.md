"""
Image Utilities — resize and crop images for WordPress thumbnails.
"""

from PIL import Image
import os


def resize_to_square(
    input_path: str,
    output_path: str = None,
    size: int = 150,
    quality: int = 85,
) -> str:
    """
    Resize and crop an image to a square thumbnail.

    Args:
        input_path: Path to the input image file
        output_path: Path to save the resized image (defaults to input_path)
        size: Target size in pixels (default: 150)
        quality: JPEG quality (1-100, default: 85)

    Returns:
        Path to the resized image file
    """
    if output_path is None:
        output_path = input_path

    # Open image
    with Image.open(input_path) as img:
        # Convert to RGB if necessary (for JPEG output)
        if img.mode != 'RGB':
            img = img.convert('RGB')

        # Calculate crop dimensions to center-crop to square
        width, height = img.size
        min_dimension = min(width, height)

        # Calculate crop box (center crop)
        left = (width - min_dimension) // 2
        top = (height - min_dimension) // 2
        right = left + min_dimension
        bottom = top + min_dimension

        # Crop to square
        img = img.crop((left, top, right, bottom))

        # Resize to target size
        img = img.resize((size, size), Image.Resampling.LANCZOS)

        # Save with appropriate format
        ext = os.path.splitext(output_path)[1].lower()
        if ext in ['.jpg', '.jpeg']:
            img.save(output_path, 'JPEG', quality=quality, optimize=True)
        elif ext == '.png':
            img.save(output_path, 'PNG', optimize=True)
        else:
            # Default to JPEG for unknown formats
            output_path = os.path.splitext(output_path)[0] + '.jpg'
            img.save(output_path, 'JPEG', quality=quality, optimize=True)

    return output_path
```

            }
        },
    )

    import httpx

    api_key = await _get_openai_key()
    client = AsyncOpenAI(api_key=api_key)

    response = await client.images.generate(
        model=model,
        prompt=prompt,
        size="256x256",  # Smallest available size for DALL-E 3

        n=1,

async def generate_image_with_gemini(
    prompt: str, save_dir: str = "/tmp/wp_images"
) -> str:
    """Generate an image using Gemini and save it locally. Returns the file path."""
    from google import genai
    from google.genai import types

    api_key = await _get_gemini_key()
    client = genai.Client(api_key=api_key)

    response = client.models.generate_content(
        model="gemini-2.0-flash-exp-image-generation",
        contents=prompt,
        config=types.GenerateContentConfig(
            response_modalities=["TEXT", "IMAGE"],
        ),
    )

    os.makedirs(save_dir, exist_ok=True)

    for part in response.candidates[0].content.parts:
        if part.inline_data and part.inline_data.mime_type.startswith("image/"):
            ext = part.inline_data.mime_type.split("/")[1]
            filename = f"{uuid.uuid4()}.{ext}"
            filepath = os.path.join(save_dir, filename)
            with open(filepath, "wb") as f:
                f.write(part.inline_data.data)

            # Resize to 150x150 square
            from app.utils.image_utils import resize_to_square
            resized_path = resize_to_square(filepath, size=150)

            return resized_path

    raise Exception("No image was generated by Gemini")
```

## Technical Details

### Image Resizing Strategy

**Center Crop + Resize:**
1. Calculate minimum dimension (width or height)
2. Center-crop to square using the minimum dimension
3. Resize to 150x150 using LANCZOS resampling (high quality)
4. Save with optimization

**Why Center Crop?**
- Maintains image quality better than stretching
- Ensures square aspect ratio
- Focuses on the center of the image (usually the most important part)

**Quality Settings:**
- JPEG quality: 85 (good balance between quality and file size)
- PNG: optimized compression
- LANCZOS resampling: highest quality for downscaling

### OpenAI DALL-E Size Options

**Available sizes for DALL-E 3:**
- 1024x1024 (standard)
- 1024x1792 (portrait)
- 1792x1024 (landscape)

**Chosen size: 256x256**
- Wait, DALL-E 3 doesn't support 256x256!
- Need to use 1024x1024 and resize down
- This is acceptable since we're resizing anyway

**Correction:** Keep size="1024x1024" and resize to 150x150

### Gemini Size Options

**Gemini 2.0 Flash:**

### Issue 2: File Size Too Large
**Mitigation:**

- 150x150 images are naturally small
- Monitor file sizes in testing

### Issue 3: Aspect Ratio Distortion
**Mitigation:**
- Use center crop to maintain aspect ratio
- Don't stretch images
- Test with extreme aspect ratios

### Issue 4: Unsupported Image Formats
**Mitigation:**
- Convert non-RGB images to RGB
- Default to JPEG for unknown formats
- Handle errors gracefully
